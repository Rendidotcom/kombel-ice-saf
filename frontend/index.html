async function downloadCertificate() { const btn = document.getElementById('btn'); const msg = document.getElementById('message'); const email = document.getElementById('email').value.trim(); msg.textContent = ''; msg.className = ''; if (!email) { msg.textContent = 'Email wajib diisi!'; msg.classList.add('error'); return; } btn.disabled = true; btn.textContent = 'Memproses...'; try { const encodedEmail = email.replace(/[^a-zA-Z0-9]/g, '_'); // fix here const fileUrl = `https://jmmqpqbpdmaelfnjhgly.supabase.co/storage/v1/object/public/certificates/${encodedEmail}.pdf`; const response = await fetch(fileUrl); if (!response.ok) { msg.textContent = 'Sertifikat untuk email ini tidak ditemukan.'; msg.classList.add('error'); return; } const blob = await response.blob(); const url = URL.createObjectURL(blob); const link = document.createElement('a'); link.href = url; link.download = `sertifikat-${email}.pdf`; document.body.appendChild(link); link.click(); document.body.removeChild(link); URL.revokeObjectURL(url); msg.textContent = 'Berhasil! Sertifikat sedang diunduh…'; msg.classList.add('success'); } catch (err) { console.error(err); msg.textContent = 'Terjadi kesalahan saat mengunduh.'; msg.classList.add('error'); } finally { btn.disabled = false; btn.textContent = 'Cek & Download'; } }
